<script>
  function initSnap() {
    const container = document.querySelector(".snap-container");
    if (!container) return;

    const sections = container.querySelectorAll(".snap-section");
    const contents = container.querySelectorAll(".snap-section .content");

    // Reset visibility (important when navigating back)
    contents.forEach((c) => c.classList.remove("visible"));

    // Fade in first section after load
    const first = sections[0]?.querySelector(".content");
    if (first) {
      setTimeout(() => first.classList.add("visible"), 200);
    }

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          const content = entry.target.querySelector(".content");
          if (!content) return;

          if (entry.isIntersecting) {
            setTimeout(() => content.classList.add("visible"), 200);
          } else {
            content.classList.remove("visible");
          }
        });
      },
      {
        root: container,
        threshold: 0.6,
      }
    );

    sections.forEach((s) => observer.observe(s));
  }

  // Run on every Astro page load (NOT just first visit)
  document.addEventListener("astro:page-load", initSnap);
</script>

<script>
  document.addEventListener("DOMContentLoaded", () => {
    const container = document.querySelector(".snap-container");
    const sections = document.querySelectorAll(".snap-section");
    const dots = document.querySelectorAll(".scroll-dots .dot");

    if (!container) return;

    // Intersection observer based on the snap-container as the scroll root
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            const id = entry.target.id;
            dots.forEach((dot) => {
              dot.classList.toggle("active", dot.dataset.target === id);
            });
          }
        });
      },
      {
        root: container,
        threshold: 0.6,
      }
    );

    sections.forEach((section) => observer.observe(section));

    // Clicking a dot scrolls the snap-container to the correct section
    dots.forEach((dot) => {
      dot.addEventListener("click", () => {
        const targetId = dot.dataset.target;
        const section = document.getElementById(targetId);
        if (!section) return;
        section.scrollIntoView({ behavior: "smooth", block: "start" });
      });
    });
  });
</script>

  window.addEventListener("DOMContentLoaded", () => {
    const canvas = document.getElementById("ascii-sagebrush");
    if (!(canvas instanceof HTMLCanvasElement)) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    const art = [
      "                 .         .         *      ",
      "        *                 .              . ",
      "    .        .    /\\      .    /\\         ",
      "      /\\      /  \\\\   /\\      /  \\\\   ",
      "_____/__\\____/____\\\\_/__\\____/____\\\\_____",
    ];

    const draw = () => {
      const rect = canvas.getBoundingClientRect();
      const dpr = window.devicePixelRatio || 1;

      canvas.width = rect.width * dpr;
      canvas.height = 200 * dpr;

      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

      ctx.fillStyle = "#F8F7F4";
      ctx.fillRect(0, 0, rect.width, 200);

      ctx.fillStyle = "#6F8F72";
      ctx.font = "14px monospace";

      const lineHeight = 20;
      const startY = 60;
      art.forEach((line, i) => {
        const metrics = ctx.measureText(line);
        const x = (rect.width - metrics.width) / 2;
        const y = startY + i * lineHeight;
        ctx.fillText(line, x, y);
      });
    };

    window.addEventListener("resize", draw);
    draw();
  });
</script>

export default class AsciiCanvas {
  canvas: HTMLCanvasElement;
  ctx: CanvasRenderingContext2D;
  width = 0;
  height = 0;

  constructor(canvas: HTMLCanvasElement) {
    this.canvas = canvas;
    this.ctx = this.canvas.getContext("2d")!;
    this.resize();
    window.addEventListener("resize", () => this.resize());
    this.render();
  }

  resize() {
    this.width = this.canvas.width = window.innerWidth;
    this.height = this.canvas.height = 200;
    this.ctx.font = "14px monospace";
  }

  render() {
    this.ctx.fillStyle = "#F8F7F4"; // background
    this.ctx.fillRect(0, 0, this.width, this.height);
    this.ctx.fillStyle = "#6F8F72"; // sagebrush green

    const art = [
      "              .        .        *        ",
      "       *           ðŸŒ¿           .     ",
      "   ðŸŒ¿        .        ðŸŒ¿             ",
      "________________________________________",
    ];

    const startY = 60;
    art.forEach((line, i) => {
      const x = (this.width - this.ctx.measureText(line).width) / 2;
      const y = startY + i * 20;
      this.ctx.fillText(line, x, y);
    });
  }
}
